\section{The ARCP Pole Leg — P1 Fig.\,5}
\label{sec:P1Fig5}
% ============================================================

\figentry{fig:P1Fig5}{P1_Fig5_ARCP_pole_leg.png}
         {The Auxiliary Resonant Commutated Pole (ARCP) single phase leg.
          Source: De Doncker \& Lyons, IEEE IAS 1990, p.\,3.}{0.75}

\begin{circuitbox}[Circuit Description — P1 Fig.\,5]
The ARCP pole leg consists of:
\begin{itemize}
    \item \textbf{DC bus}: split into two equal halves by two $2C_{dc}$
          capacitors, creating a midpoint rail at $\Vdc/2$
    \item \textbf{Main upper device}: $S_1$ (IGBT) with anti-parallel diode
          $D_1$; snubber capacitor $\Cr/2$ across $S_1$
    \item \textbf{Main lower device}: $S_2$ (IGBT) with anti-parallel diode
          $D_2$; snubber capacitor $\Cr/2$ across $S_2$
    \item \textbf{Pole node}: the junction of $S_1$ and $S_2$; voltage $\vf$
          swings between 0 and $\Vdc$ during commutation
    \item \textbf{Auxiliary switches}: $A_1$ and $A_2$ are back-to-back
          bidirectional switches (p-channel MCTs) connected between the
          \emph{DC bus midpoint} and the resonant inductor $\Lr$
    \item \textbf{Resonant inductor}: $\Lr$ connects from the $A_1$/$A_2$
          pair to the pole node; carries auxiliary current $\ir$
    \item \textbf{Load}: filter inductor $L_F$ and load current $I_F$ at
          the pole node; $V_F$ is the filtered output voltage
\end{itemize}
The auxiliary branch ($A_1$--$A_2$--$\Lr$) is \emph{only active during
commutation}. At steady state $\ir = 0$.
\end{circuitbox}

\begin{eqnbox}
\textbf{When do these equations apply? — the exact circuit condition.}

These equations are valid only during the brief interval when all of the
following are simultaneously true:
\begin{itemize}
    \item $A_1$ is \textbf{on} — the DC midpoint is connected through $\Lr$
          to the pole node, so $\Vdc/2$ drives the auxiliary inductor.
    \item $D_2$ has \textbf{just stopped conducting} — the pole node is no
          longer clamped to 0\,V and is free to rise.
    \item Both $\Cr/2$ snubber capacitors are free to charge and discharge —
          no main switch is actively short-circuiting the pole node.
\end{itemize}
The instant any one of those conditions changes (e.g.\ $D_1$ clamps, or
$A_1$ turns off) the circuit topology changes and different equations take
over.

\medskip
\textbf{The two circuit equations for this condition.}

\textit{KCL at the pole node:}
$\ir$ flows in; $\IL$ flows out to the load; the rest charges the combined
snubber capacitance $\Cr$ (= $\Cr/2$ in parallel with $\Cr/2$, seen from
the pole node):
\begin{equation}
    \frac{d\vf}{dt} = \frac{\ir - \IL}{\Cr}
    \label{eq:dvf_dt}
\end{equation}

\textit{KVL around the auxiliary loop ($A_1$--$\Lr$--pole\,node):}
The DC midpoint sits at $\Vdc/2$.
The pole node sits at $\vf$.
The full voltage difference $(\Vdc/2 - \vf)$ appears across $\Lr$ and
drives the rate of change of $\ir$:
\begin{equation}
    \frac{d\ir}{dt} = \frac{\Vdc/2 - \vf}{\Lr}
    \label{eq:dir_dt}
\end{equation}

\medskip
\textbf{Step 1 — Combine into one equation for $\vf$ alone.}

Equation~\eqref{eq:dvf_dt} involves both $\vf$ and $\ir$.
Equation~\eqref{eq:dir_dt} tells us how $\ir$ changes.
To get a single equation in $\vf$ only, differentiate
\eqref{eq:dvf_dt} once with respect to time:
\begin{equation*}
    \frac{d^2\vf}{dt^2} = \frac{1}{\Cr}\,\frac{d\ir}{dt}
\end{equation*}
The right-hand side contains $d\ir/dt$, which is exactly what
\eqref{eq:dir_dt} gives us.  Substitute:
\begin{equation*}
    \frac{d^2\vf}{dt^2}
        = \frac{1}{\Cr}\cdot\frac{\Vdc/2 - \vf}{\Lr}
        = \frac{\Vdc/2}{\Lr\Cr} - \frac{\vf}{\Lr\Cr}
\end{equation*}
Move the $\vf$ term to the left.  Define $\omegar \triangleq 1/\sqrt{\Lr\Cr}$
so that $\omegar^2 = 1/(\Lr\Cr)$:
\begin{equation}
    \ddot{\vf} + \omegar^2\,\vf = \omegar^2\cdot\frac{\Vdc}{2}
    \label{eq:oscillator}
\end{equation}
This is a \emph{forced harmonic oscillator} — the same form as a
spring-mass system driven by a constant force.  The right-hand side acts
as the ``equilibrium point'': $\vf$ wants to settle at $\Vdc/2$
(the DC midpoint), not at zero.

\medskip
\textbf{Step 2 — Simplify by measuring from the equilibrium point.}

The equilibrium of \eqref{eq:oscillator} sits at $\vf = \Vdc/2$.
Working directly with $\vf$ keeps dragging this constant around.
A cleaner approach is to measure $\vf$ \emph{relative to} the equilibrium.
Define:
\begin{equation*}
    u \triangleq \vf - \frac{\Vdc}{2}
    \qquad (\text{deviation of pole voltage from DC midpoint})
\end{equation*}
Also, notice that $\IL$ is a constant (load current changes slowly compared
to the commutation), so define:
\begin{equation*}
    j \triangleq \ir - \IL
    \qquad (\text{auxiliary current above the load current})
\end{equation*}
Since $\IL$ is constant, $\dot{j} = \dot{\ir}$.
Now rewrite \eqref{eq:dvf_dt} and \eqref{eq:dir_dt} in terms of $u$ and $j$:
\begin{align*}
    \dot{u} = \dot{\vf}
        &= \frac{\ir - \IL}{\Cr} = \frac{j}{\Cr} \\[6pt]
    \dot{j} = \dot{\ir}
        &= \frac{\Vdc/2 - \vf}{\Lr}
         = \frac{\Vdc/2 - (u + \Vdc/2)}{\Lr}
         = \frac{-u}{\Lr}
\end{align*}
The constant $\Vdc/2$ has dropped out completely.
Defining $\Zr \triangleq \sqrt{\Lr/\Cr}$ (the characteristic impedance of
the LC circuit), the final system is:
\begin{equation}
    \dot{u} = \frac{j}{\Cr}, \qquad \dot{j} = -\frac{u}{\Lr}
    \label{eq:shifted_states}
\end{equation}
These two equations are symmetric and coupled: $u$ drives $j$, and
$j$ drives $u$ back. This back-and-forth is the hallmark of oscillation.

\medskip
\textbf{Step 3 — What shape does the trajectory trace?}

From Step~1 we already know that $u$ satisfies
$\ddot{u} + \omegar^2 u = 0$ — simple harmonic motion (SHM), the same
equation as a mass on a spring with no friction.
Every student of SHM knows that the general solution is a sinusoid:
\begin{equation*}
    u(t) = u_0 \cos(\omegar t) + C\,\sin(\omegar t)
\end{equation*}
where $u_0 = u(0)$ is fixed by the initial condition and $C$ is determined
by the initial velocity.
From \eqref{eq:shifted_states}, $\dot{u} = j/\Cr$, so at $t=0$:
\begin{equation*}
    \dot{u}(0) = \frac{j_0}{\Cr}
    \quad\Longrightarrow\quad
    C\,\omegar = \frac{j_0}{\Cr}
    \quad\Longrightarrow\quad
    C = \frac{j_0}{\Cr\,\omegar}
\end{equation*}
Now $\Cr\,\omegar = \Cr/\sqrt{\Lr\Cr} = \sqrt{\Cr/\Lr} = 1/\Zr$,
so $C = j_0\Zr$.  The explicit solution for $u(t)$ is:
\begin{equation}
    u(t) = u_0\cos(\omegar t) + j_0\Zr\,\sin(\omegar t)
    \label{eq:u_solution}
\end{equation}

Now get $j(t)$ directly from \eqref{eq:shifted_states}: $j = \Cr\dot{u}$.
Differentiate \eqref{eq:u_solution} and multiply by $\Cr$:
\begin{align*}
    j(t) &= \Cr\bigl[-u_0\,\omegar\sin(\omegar t)
                      + j_0\Zr\,\omegar\cos(\omegar t)\bigr] \\[4pt]
         &= \Cr\,\omegar\bigl[-u_0\sin(\omegar t)
                               + j_0\Zr\cos(\omegar t)\bigr]
\end{align*}
Using $\Cr\,\omegar = 1/\Zr$ again:
\begin{equation}
    j(t) = -\frac{u_0}{\Zr}\sin(\omegar t) + j_0\cos(\omegar t)
    \label{eq:j_solution}
\end{equation}

\medskip
We now have explicit expressions for both $u(t)$ and $j(t)$.
Define two new coordinates:
\begin{equation*}
    X \triangleq u, \qquad Y \triangleq j\,\Zr
\end{equation*}
($Y$ is simply $j$ rescaled so that both axes carry the same units of
volts.)  Writing out $X$ and $Y$:
\begin{align*}
    X &= u_0\cos(\omegar t) + j_0\Zr\,\sin(\omegar t) \\[4pt]
    Y &= j\Zr = -u_0\sin(\omegar t) + j_0\Zr\,\cos(\omegar t)
\end{align*}
Now compute $X^2 + Y^2$, expanding term by term:
\begin{align*}
    X^2 &= u_0^2\cos^2(\omegar t)
           + 2u_0 j_0\Zr\cos(\omegar t)\sin(\omegar t)
           + j_0^2\Zr^2\sin^2(\omegar t) \\[4pt]
    Y^2 &= u_0^2\sin^2(\omegar t)
           - 2u_0 j_0\Zr\sin(\omegar t)\cos(\omegar t)
           + j_0^2\Zr^2\cos^2(\omegar t)
\end{align*}
Adding: the cross terms ($\pm 2u_0 j_0\Zr\cos\sin$) cancel exactly, and
$\cos^2 + \sin^2 = 1$ collapses each remaining pair:
\begin{equation}
    X^2 + Y^2
        = u_0^2\underbrace{(\cos^2 + \sin^2)}_{1}
        + j_0^2\Zr^2\underbrace{(\sin^2 + \cos^2)}_{1}
        = u_0^2 + (j_0\Zr)^2
    \label{eq:circle}
\end{equation}
The right-hand side is a \emph{constant} (set by the initial conditions,
not by time).  Therefore, as $t$ increases, the point $(X, Y) = (u,\,j\Zr)$
moves along a \textbf{circle} of radius:
\begin{equation*}
    R = \sqrt{u_0^2 + (j_0\Zr)^2}
\end{equation*}
centred at the origin of the $(u,\,j\Zr)$ plane, which is the point
$(\vf,\,j\Zr) = (\Vdc/2,\,0)$ in the original coordinates.

\medskip
\textbf{Initial conditions and the radius of the circle.}

Just before the resonant swing begins, $D_2$ is still conducting and
pins $\vf = 0$.  The boost phase has already ramped $\ir$ up to
$\IL + i_\text{boost}$.  So:
\begin{equation}
    u_0 = 0 - \frac{\Vdc}{2} = -\frac{\Vdc}{2},
    \qquad
    j_0 = (\IL + i_\text{boost}) - \IL = i_\text{boost}
    \label{eq:swing_ic}
\end{equation}
Substituting into \eqref{eq:circle}:
\begin{equation}
    R = \sqrt{\left(\frac{\Vdc}{2}\right)^{\!2} + (i_\text{boost}\,\Zr)^2}
    \label{eq:radius}
\end{equation}
\textbf{How does the phase plane connect to time?}

The phase plane has no time axis, so it is not obvious how a longer or
shorter arc corresponds to a longer or shorter commutation.
The answer comes directly from the parametric solution
\eqref{eq:u_solution}--\eqref{eq:j_solution}, which we can write as:
\begin{equation*}
    X(t) = u(t), \qquad Y(t) = j(t)\Zr,
    \qquad \text{where the parameter is } \theta = \omegar t
\end{equation*}
Every time step $\Delta t$ advances the angle $\theta$ by exactly
$\omegar\,\Delta t$, regardless of where on the circle the trajectory
currently sits.
Therefore the time taken to travel from the start angle $\theta_\text{start}$
to the end angle $\theta_\text{end}$ is:
\begin{equation*}
    T_\text{comm} = \frac{\theta_\text{end} - \theta_\text{start}}{\omegar}
\end{equation*}
A smaller swept angle means a shorter commutation time.

\textbf{How does $i_\text{boost}$ change the swept angle?}

All three trajectories in \figref{fig:phase_plane_3ICs} start at the
same $\vf = 0$ (i.e.\ $u_0 = -\Vdc/2$) but at different heights
$j_0\Zr = i_\text{boost}\Zr$ on the vertical axis.
The target is always $\vf = \Vdc$ (i.e.\ $u = +\Vdc/2$).
A higher starting point is already closer (in angle) to the target on the
right side of the circle, so less angle needs to be swept.
The extreme case is $i_\text{boost} = 0$: the trajectory starts exactly on
the horizontal axis at $(-\Vdc/2,\;0)$ and must sweep a full
$\pi$\,radians (a semicircle) to reach $(+\Vdc/2,\;0)$, taking time
$\pi/\omegar$.
Every additional ampere of boost lifts the starting point, reduces the
angle to be swept, and shortens the commutation time.
\end{eqnbox}

\begin{physbox}
% ── 1. Why ZVS eliminates switching loss ─────────────────────────────────────
\textbf{1.\ Why ZVS eliminates switching loss}

Consider closing any switch (here an IGBT $S_1$) in a circuit.
The switch transitions from fully off to fully on in a short but finite time
$t_\text{sw}$.  During that transition:
\begin{itemize}
    \item there is a voltage $V_{S_1}$ across the switch (it has not yet
          fully turned on), \emph{and}
    \item there is a current $I_{S_1}$ flowing through it (the load is
          already demanding current).
\end{itemize}
Instantaneous power dissipated in the switch $= V_{S_1} \times I_{S_1}$.
Integrated over the transition, this gives an \textbf{energy loss every
single switching cycle}.  At high switching frequencies (tens of kHz to MHz)
these losses add up and heat the switch, reducing efficiency and requiring
large heatsinks.

\textbf{Zero-voltage switching} eliminates this loss by a simple idea:
\emph{ensure the voltage across $S_1$ is zero at the exact moment you
fire it.}  Then $P = 0 \times I_{S_1} = 0$ regardless of how large the
current is — the switch turns on for free.

\medskip
\textbf{How does the ARCP achieve ZVS for $S_1$?}

The voltage across $S_1$ is
\begin{equation*}
    V_{S_1} = \Vdc - \vf
\end{equation*}
because the upper terminal of $S_1$ sits at $\Vdc$ and the lower terminal
(the pole node) sits at $\vf$.  For $V_{S_1} = 0$ we need
$\vf = \Vdc$.

So the entire job of the resonant swing is:
\emph{drive $\vf$ from 0 all the way up to $\Vdc$ before $S_1$ is fired.}
Once $\vf$ reaches $\Vdc$, the diode $D_1$ clamps it there — $D_1$ is the
anti-parallel diode of $S_1$, with its cathode at the positive rail ($\Vdc$)
and its anode at the pole node ($\vf$).
The moment $\vf$ touches $\Vdc$, $D_1$ becomes forward-biased and starts
conducting, diverting $\ir$ away from \emph{both} $\Cr/2$ snubber capacitors
and into the positive DC rail directly.
To see why both are involved: during the resonant swing $\ir$ was
simultaneously charging the $\Cr/2$ across $S_2$ (from 0 up to $\Vdc$)
\emph{and} discharging the $\Cr/2$ across $S_1$ (from $\Vdc$ down to 0).
When $\vf = \Vdc$, both processes complete at the same instant —
the $\Cr/2$ across $S_2$ is fully charged, and the $\Cr/2$ across $S_1$
is fully discharged to zero.
$D_1$ conducts and locks $\vf$ at $\Vdc$, so neither cap can change
further; $\ir$ flows through $D_1$ into the bus instead.
$S_1$ now has $V_{S_1} = \Vdc - \vf = 0$ and can be gated on with no
switching loss.

\medskip
% ── 2. Why R >= Vdc/2 is the ZVS geometric condition ─────────────────────────
\textbf{2.\ Why $R \geq \Vdc/2$ is the ZVS geometric condition}

We proved that the resonant swing traces a circle in the
$(\vf,\,j\Zr)$ plane with:
\begin{itemize}
    \item \textbf{Centre} at $\vf = \Vdc/2$ (the DC midpoint)
    \item \textbf{Radius} $R = \sqrt{({\Vdc}/{2})^2 + (j_0\Zr)^2}$
\end{itemize}
The rightmost point the trajectory can reach is
\begin{equation*}
    \vf^\text{max} = \frac{\Vdc}{2} + R
\end{equation*}
For $\vf$ to reach $\Vdc$ (the ZVS target), we need:
\begin{equation*}
    \frac{\Vdc}{2} + R \geq \Vdc
    \quad\Longrightarrow\quad
    R \geq \frac{\Vdc}{2}
\end{equation*}
Substituting the radius formula:
\begin{equation*}
    \sqrt{\left(\frac{\Vdc}{2}\right)^{\!2} + (j_0\Zr)^2} \geq \frac{\Vdc}{2}
\end{equation*}
This is automatically true for \emph{any} $j_0 \geq 0$, because
$\sqrt{a^2 + b^2} \geq a$ whenever $b \geq 0$.

\medskip
\textbf{What goes wrong when $j_0 < 0$?}

$j_0 = \ir(0) - \IL$ is the auxiliary current \emph{above} the load current
at the start of the swing.  If $j_0 < 0$ (i.e.\ $\ir < \IL$), the net current
charging $\Cr$ is negative: from equation~\eqref{eq:dvf_dt},
$\dot{\vf} = j_0/\Cr < 0$.
The pole voltage \emph{falls} initially instead of rising.
In the phase plane this means the trajectory starts going in the wrong
direction — downward and to the left — before eventually looping back.
Physically, diode $D_2$ immediately clamps $\vf$ at 0 and prevents
it from going negative, so the commutation \emph{fails}.
This is visible in the IC-1 trajectory ($j_0=0$) of
Figure~\ref{fig:phase_plane_3ICs}: the trajectory starts horizontally
(neither rising nor falling), which is the marginal case.

\medskip
% ── 3. Why a boost phase is needed ───────────────────────────────────────────
\textbf{3.\ Why a boost phase is needed}

At light load the current $\IL$ is small.
When $S_2$ is turned off, the only current available to charge $\Cr$ and
raise $\vf$ is $\IL$ itself ($\ir = 0$ initially).  If $\IL$ is too small,
$\vf$ rises too slowly and cannot reach $\Vdc$ within the available
commutation window — $S_1$ would have to be fired before $\vf = \Vdc$,
causing hard switching and energy loss.

The boost phase solves this by pre-charging the resonant inductor $\Lr$:
\begin{enumerate}
    \item $A_1$ fires briefly; $\ir$ ramps up linearly from 0 to
          $\IL + i_\text{boost}$.
    \item $A_1$ turns off; the swing begins with
          $j_0 = i_\text{boost} > 0$.
    \item A positive $j_0$ gives a radius $R > \Vdc/2$, the trajectory
          goes in the right direction, and $\vf$ reaches $\Vdc$ — ZVS
          is achieved even at light load.
\end{enumerate}
The larger $i_\text{boost}$ is, the larger $R$ becomes, and the more
quickly $\vf$ swings to $\Vdc$.  At heavy load ($\IL$ large), the load
current alone is sufficient ($j_0 = 0$ is the marginal case) and no
boost is needed — this is called \emph{natural commutation}.

\medskip
% ── 4. Cr design trade-off: commutation energy vs turn-off stress ────────────
\textbf{4.\ $\Cr$ design trade-off: commutation energy vs.\ turn-off $dv/dt$ stress}

\textit{Reason to make $\Cr$ small — reduce commutation energy:}
Each time the pole commutates, $\Cr$ must be fully charged from 0 to
$\Vdc$ (or discharged from $\Vdc$ to 0).
The energy moved in and out of $\Cr$ every commutation is
\begin{equation*}
    E_\Cr = \tfrac{1}{2}\Cr\Vdc^2
\end{equation*}
This energy is supplied by the auxiliary inductor $\Lr$ (via the boost
and resonant swing) and ultimately comes from the DC bus.
Smaller $\Cr$ means less energy wasted on every switching event.

\textit{Reason not to make $\Cr$ too small — limit the voltage slew rate:}
When $S_2$ turns off, its current does not disappear instantly — it
diverts into $\Cr$ and begins charging it.
The rate at which the voltage across $S_2$ rises is
\begin{equation*}
    \frac{d\vf}{dt} = \frac{\ir - \IL}{\Cr}
\end{equation*}
If $\Cr$ is very small, this rate becomes very large — the voltage
across $S_2$ spikes up extremely fast.
Real IGBTs and MOSFETs have a finite tolerance for how fast their
blocking voltage can rise ($dv/dt$ rating).
Exceeding it causes extra losses, gate-ringing, or device failure.

\textit{Summary — the design trade-off:}
Choose $\Cr$ small enough that $E_\Cr = \frac{1}{2}\Cr\Vdc^2$ is
acceptably small, yet large enough that the turn-off $dv/dt$ stays
within the switch's rating.
\end{physbox}

\section{Commutation from a Conducting Diode — P1 Fig.\,7}
\label{sec:P1Fig7}

\subsection*{Big picture — what is Fig.\,7 teaching?}

An inverter pole-leg has one job: connect the output node (the pole node,
$\vf$) to either the positive rail ($\Vdc$) or the negative rail (0\,V).
Switching between these two states is called \textbf{commutation}.

In a hard-switched inverter you would simply turn off the lower device and
fire the upper device.  That sounds simple, but during the brief transition
the switch carries both a large voltage \emph{and} a large current
simultaneously, dissipating energy as heat on every single switching cycle.

Fig.\,7 answers the question: \emph{how does the ARCP solve this, step by
step, starting from the moment $D_2$ is conducting?}

The word ``from diode'' describes the \emph{starting condition}: the load
current $\IL$ is flowing upward through the pole node into the load, and the
only path available for this current on the lower side is the anti-parallel
diode $D_2$.  The converter must now move $\vf$ from 0 all the way up to
$\Vdc$ so that the upper switch $S_1$ can be turned on at zero voltage
(ZVS).  Fig.\,7 shows the exact sequence of seven circuit states ---
(a)\,through\,(g) --- that accomplish this safely.

\medskip
\noindent
\textbf{How each state is drawn.}
Every sub-figure in Fig.\,7 is a pair of two panels:
\begin{itemize}
    \item \textbf{Top panel --- the waveform strip.}
          This shows $\ir(t)$ and $\vf(t)$ on a shared time axis covering
          the \emph{entire} commutation from state~(a) to state~(g).
          Vertical dashed lines divide the axis into strips; each strip is
          labelled at the bottom with the device(s) that are conducting
          during that interval (e.g.\ \textit{D2}, \textit{D2/A2},
          \textit{Cr/A2}, \textit{S1}).
          A horizontal reference line on the $\ir$ plot marks the load
          current level $\IL$.
          The sub-figure letter tells you which strip you are currently in.

    \item \textbf{Bottom panel --- the circuit snapshot.}
          This strips the full ARCP schematic down to only the devices
          that are conducting right now; their current paths are drawn in
          bold.  Everything else is open.  It is a ``frozen frame'' of the
          circuit topology for the labelled time strip.
\end{itemize}

Reading both panels together for each state builds the complete physical
picture: you see \emph{where in time} you are (waveform) and
\emph{which path the current takes} (circuit snapshot) simultaneously.

% ── State (a) explanation ─────────────────────────────────────────────────────
\subsection*{State (a) — $D_2$ conducting: the resting state before commutation}

\figentry{fig:P1Fig7a}{P1_Fig7a_state1_D2_conducting}
    {P1 Fig.\,7(a) — $D_2$ conducting: initial resting state.
     $\vf = 0$, $\ir = 0$.
     Top panel: full commutation waveform with state~(a) strip highlighted.
     Bottom panel: circuit snapshot showing $D_2$ as the only active path.
     Source: De~Doncker \& Lyons, IEEE IAS 1990, p.\,4.}{0.50}

\textbf{Top panel (waveform) — what you see in the state\,(a) strip.}

State~(a) occupies the leftmost strip of the waveform, labelled \textit{D2}.
Both traces are completely flat throughout this strip:
\begin{itemize}
    \item $\ir = 0$\,A — no current flows in the resonant branch at all.
          The auxiliary inductor $\Lr$ is at rest.
    \item $\vf = 0$\,V — the pole node is pinned to the negative DC rail.
\end{itemize}
This strip has no fixed duration.  It lasts as long as the PWM lower-side
conduction period lasts — anywhere from a few microseconds to tens of
microseconds depending on the modulation index and load.
What matters is not how long state~(a) lasts, but the two numbers it
establishes before commutation begins:
\begin{equation}
    \vf(0^-) = 0\,\text{V}, \qquad \ir(0^-) = 0\,\text{A}
    \label{eq:stateA_ic}
\end{equation}
These are the initial conditions that every subsequent calculation in states
(b)\,through\,(g) is built on.

\medskip
\textbf{Bottom panel (circuit snapshot) — what is conducting and why.}

Only $D_2$ appears with a bold current path; all other devices are open.
The load current $\IL$ flows along the path:
\[
    \text{negative DC rail} \;\longrightarrow\; D_2 \;\longrightarrow\;
    \text{pole node} \;\longrightarrow\; L_F \;\longrightarrow\; \text{load}
\]
$D_2$ is not actively driven by any gate signal — it is a diode.
It conducts simply because $\IL$ pushes current in the upward direction
and $D_2$ is the only forward-biased path on the lower side of the pole.
The two snubber capacitors ($\Cr/2$ across $S_1$ and $\Cr/2$ across $S_2$)
sit at fixed voltages: the one across $S_1$ is charged to $\Vdc$
(full bus voltage); the one across $S_2$ is discharged to 0\,V
(short-circuited by the conducting $D_2$).
The main switches $S_1$, $S_2$ and both auxiliary back-to-back switches
$A_1$, $A_2$ are all open.

\medskip
\textbf{Why this state exists — what it does for the system.}

State~(a) is the \emph{resting state between switching events}.
Its role is to guarantee that when a commutation is eventually triggered,
the circuit starts from a fully known, clean condition
(equation~\eqref{eq:stateA_ic}).
If $\ir$ were not zero at the start — for example, if a previous
commutation had not fully reset the auxiliary branch — the resonant swing
would start from a different initial point and the ZVS condition might fail.
State~(a) ensures the slate is wiped clean every cycle.

% ── State (b) explanation ─────────────────────────────────────────────────────
\subsection*{State (b) — $A_2$ fires: the boost phase begins}

\figentry{fig:P1Fig7b}{P1_Fig7b_state2_A1_boost_start}
    {P1 Fig.\,7(b) — $A_2$ fires: $\ir$ begins ramping linearly from zero.
     $D_2$ still conducts and keeps $\vf = 0$.
     Top panel: $\ir$ strip rising linearly in the \textit{D2/A2} region.
     Bottom panel: bold paths show $A_2$ and $D_2$ both conducting.
     Source: De~Doncker \& Lyons, IEEE IAS 1990, p.\,4.}{0.50}

\textbf{What triggers this state.}

The control logic decides it is time to commutate from $D_2$ to $S_1$.
It fires the auxiliary switch $A_2$ --- the lower half of the back-to-back
bidirectional switch pair on the left of the pole leg
(see \figref{fig:P1Fig7b}, bottom panel).

\medskip
\textbf{Top panel (waveform) — what you see in the state\,(b) strip.}

We are now in the second strip, labelled \textit{D2/A2} in the waveform of
\figref{fig:P1Fig7b}.
\begin{itemize}
    \item $\ir$ starts rising from 0 in a straight line --- a \emph{linear
          ramp}. The slope is constant throughout this strip.
    \item $\vf$ is still completely flat at 0\,V.
\end{itemize}

\medskip
\textbf{Bottom panel (circuit snapshot) — two devices conducting at once.}

Two current paths are active simultaneously (both drawn bold):
\begin{enumerate}
    \item \textbf{The new $A_2$--$\Lr$ path:}
          $A_2$ connects the DC bus midpoint (held at $\Vdc/2$ by the two
          $2C_{dc}$ capacitors) to one terminal of $\Lr$.
          The other terminal of $\Lr$ connects to the pole node, which is
          still clamped at 0\,V by $D_2$.
          So the full voltage $\Vdc/2 - 0 = \Vdc/2$ appears across $\Lr$,
          driving $\ir$ to ramp upward.

    \item \textbf{The continuing $D_2$--load path:}
          The load current $\IL$ still needs a return path to the negative
          rail.  $D_2$ provides it, carrying the \emph{difference}
          $\IL - \ir$ (which decreases as $\ir$ rises).
          While $D_2$ conducts, it clamps $\vf$ firmly at 0\,V.
          This in turn keeps the full $\Vdc/2$ across $\Lr$, maintaining
          the constant ramp rate.
\end{enumerate}

\medskip
\textbf{Why the ramp is linear — the inductor equation.}

An inductor obeys $V_{\Lr} = \Lr\,\dfrac{d\ir}{dt}$.
Rearranging:
\begin{equation}
    \frac{d\ir}{dt} = \frac{V_{\Lr}}{\Lr} = \frac{\Vdc/2}{\Lr}
                    = \frac{\Vdc}{2\Lr}
    \label{eq:boost_ramp}
\end{equation}
The right-hand side is a constant (both $\Vdc$ and $\Lr$ are fixed).
A constant $d\ir/dt$ means $\ir$ increases at a \emph{constant rate} ---
a straight line on the waveform.
Starting from $\ir(0^-) = 0$:
\begin{equation}
    \ir(t) = \frac{\Vdc}{2\Lr}\,t
    \label{eq:ir_ramp}
\end{equation}

\medskip
\textbf{Why $\Vdc/2$ and not $\Vdc$?}

The auxiliary branch is connected to the \emph{DC bus midpoint}, not to
the full positive rail.  The midpoint is created by two $2C_{dc}$
capacitors that split the bus exactly in half.
Using the midpoint halves the driving voltage, which halves the ramp rate
and gives the control system more time to hit its current target precisely.
If the full $\Vdc$ were used, $\ir$ would ramp twice as fast and the timing
would be harder to control.

\medskip
\textbf{What ends this state.}

State~(b) ends when $\ir$ rises to exactly $\IL$.
At that point the current $D_2$ must supply becomes zero ($\IL - \ir = 0$),
and $D_2$ turns off \emph{naturally} --- no gate signal is needed, the
diode simply stops conducting because it has no more current to carry.
$A_2$ is \emph{not} turned off here; it stays on.
The control system detects (or predicts) this moment and immediately
fires $S_2$, which takes over from $D_2$ and keeps $\vf = 0$.
The circuit then enters state~(c).

% ── State (c) explanation ─────────────────────────────────────────────────────
\subsection*{State (c) — boost complete: $\ir = \IL + i_\text{boost}$, resonant swing about to begin}

\figentry{fig:P1Fig7c}{P1_Fig7c_state3_boost_complete}
    {P1 Fig.\,7(c) — Boost complete.
     $\ir$ has reached $\IL + i_\text{boost}$; $\vf$ is still 0.
     $A_2$ and $S_2$ are both conducting; $D_2$ is off.
     Source: De~Doncker \& Lyons, IEEE IAS 1990, p.\,4.}{0.50}

\textbf{Top panel (waveform) — what you see in the state\,(c) strip.}

We are now in the third strip of the waveform (\figref{fig:P1Fig7c}),
labelled \textit{S2/A2}.
\begin{itemize}
    \item $\ir$ is \emph{still rising} linearly, continuing the ramp that
          started in state~(b), but now from $\IL$ toward
          $\IL + i_\text{boost}$.
          The slope is the same constant value $\Vdc/(2\Lr)$ as before,
          because $A_2$ is still on and $\vf$ is still held at 0\,V.
    \item $\vf$ remains flat at 0\,V --- clamped as before, just by a
          different device (see circuit snapshot below).
\end{itemize}
The strip ends when $\ir = \IL + i_\text{boost}$ and $S_2$ is turned off.

\medskip
\textbf{Bottom panel (circuit snapshot) — why did $D_2$ disappear and why is $S_2$ here?}

Two devices are bold in \figref{fig:P1Fig7c}: $A_2$ and $S_2$.
$D_2$ is no longer active.  Here is why each device changed role.

\medskip
\textbf{Why $D_2$ dropped out.}

$D_2$ is a diode.  A diode can only push current in one direction.
In state~(b), $D_2$ was supplying the \emph{deficit} current
$\IL - \ir > 0$ --- current flowing from the 0\,V rail into the pole node
to make up the difference between $\IL$ and the smaller $\ir$.
As $\ir$ climbed toward $\IL$, this deficit shrank.
The moment $\ir = \IL$, the deficit reached zero and $D_2$ had nothing
left to supply.  A diode carrying zero current simply turns off; there is
no gate signal involved --- it stops conducting on its own.

\medskip
\textbf{Why $S_2$ is fired.}

Once $D_2$ turns off, nothing is holding $\vf$ at 0\,V.
With $A_2$ still on and $\ir$ still flowing, the excess current
$\ir - \IL > 0$ would immediately start charging the snubber
capacitors and lifting $\vf$.
If we let that happen now, the resonant swing would begin with
$j_0 = \ir - \IL \approx 0$ --- barely above the marginal ZVS threshold.

To get a larger margin of safety --- a larger circle radius $R$ in the
phase plane --- we want $j_0 = i_\text{boost} > 0$ at the start of the swing.
The control achieves this by firing $S_2$ the instant $D_2$ turns off.

$S_2$ is an IGBT: an actively controlled switch that can carry current in
\emph{both} directions (unlike the diode $D_2$).
With $S_2$ on, the pole node is held at 0\,V by the switch, not the diode.
The excess current $\ir - \IL$ now flows \emph{down} through $S_2$ into
the negative rail (as shown by the downward arrow on $S_2$ in the circuit
snapshot).
This allows the boost to continue past $\IL$ while $\vf$ stays at 0\,V.

\medskip
\textbf{The full picture at the end of state (c).}

By the time $\ir$ reaches $\IL + i_\text{boost}$, we have:
\begin{equation}
    \vf = 0\,\text{V} \qquad \text{and} \qquad
    \ir = \IL + i_\text{boost}
    \label{eq:stateC_ic}
\end{equation}
These are the \emph{initial conditions for the resonant swing}.
In the shifted coordinates of the phase plane:
\begin{equation*}
    u_0 = \vf - \frac{\Vdc}{2} = -\frac{\Vdc}{2}, \qquad
    j_0 = \ir - \IL = i_\text{boost}
\end{equation*}
giving a circle radius $R = \sqrt{(\Vdc/2)^2 + (i_\text{boost}\,\Zr)^2}$,
which is larger the more boost current we add.

\medskip
\textbf{What ends this state — and how the resonant swing is triggered.}

\textit{Step 1 — What the control does.}

The control monitors $\ir$ and turns off $S_2$ the moment
$\ir = \IL + i_\text{boost}$ is reached.
$A_2$ stays on.

\medskip
\textit{Step 2 — Where was the current going before $S_2$ opened?}

Just before $S_2$ turns off, the current balance at the pole node is:
\begin{equation*}
    \underbrace{\ir}_{\text{in from } \Lr}
    \;=\;
    \underbrace{\IL}_{\text{out to load}}
    \;+\;
    \underbrace{(\ir - \IL)}_{\text{down through } S_2 \text{ to lower rail}}
\end{equation*}
$S_2$ is carrying the small excess $\ir - \IL = i_\text{boost}$ downward.
Everything is in balance and $\vf = 0$.

\medskip
\textit{Step 3 — What happens the instant $S_2$ opens?}

An open switch carries zero current.
The $i_\text{boost}$ that was flowing through $S_2$ \emph{must go somewhere}
--- current in an inductor cannot disappear instantaneously.
With $S_2$ open and $D_2$ already off, the only paths left at the pole node
are the load ($\IL$) and the two snubber capacitors.

\medskip
\textit{Why the effective capacitance at the pole node is the full $\Cr$,
not $\Cr/2$.}

When $S_2$ opens, the pole node is connected to two capacitors:
\begin{itemize}
    \item $\Cr/2$ between the pole node and the \textbf{0\,V rail}
          (the snubber across $S_2$)
    \item $\Cr/2$ between the pole node and the \textbf{$\Vdc$ rail}
          (the snubber across $S_1$)
\end{itemize}
Both rails --- 0\,V and $\Vdc$ --- are held at \emph{fixed voltages} by
the large $2C_{dc}$ bus capacitors.
No matter what happens at the pole node, these rails do not move.
In circuit terms, both rails are \textbf{AC grounds}.

From the pole node's point of view, therefore, both $\Cr/2$ capacitors
are connected between the pole node and a fixed reference.
This makes them \textbf{effectively in parallel}:
\begin{equation*}
    C_\text{eff} = \frac{\Cr}{2} + \frac{\Cr}{2} = \Cr
\end{equation*}
When $\vf$ rises, the pole node must supply current to \emph{both}
capacitors simultaneously --- one charges toward $\Vdc$, the other
discharges toward 0\,V --- but both demand current from the same pole node.
The KCL at the pole node is therefore:
\begin{equation}
    \ir \;=\; \IL
        + \underbrace{\frac{\Cr}{2}\frac{d\vf}{dt}}_{\text{into }
              \Cr/2\text{ across }S_2}
        + \underbrace{\frac{\Cr}{2}\frac{d\vf}{dt}}_{\text{into }
              \Cr/2\text{ across }S_1}
    \;=\; \IL + \Cr\frac{d\vf}{dt}
    \label{eq:kcl_pole}
\end{equation}
Rearranging and evaluating at $t = 0^+$, where $\ir$ is still
$\IL + i_\text{boost}$ (inductor current cannot jump instantaneously):
\begin{equation}
    \Cr\,\frac{d\vf}{dt}\bigg|_{t=0^+}
        = \ir - \IL = i_\text{boost}
    \quad\Longrightarrow\quad
    \frac{d\vf}{dt}\bigg|_{t=0^+} = \frac{i_\text{boost}}{\Cr}
    \label{eq:vf_initial_slope}
\end{equation}
$\vf$ immediately starts rising from 0 at a rate set by $i_\text{boost}$
and the \emph{total} pole capacitance $\Cr$.
This is the first instant of the resonant LC swing --- state~(d) has begun.

\medskip
\textit{Step 4 — Why is the turn-off of $S_2$ relatively clean?}

In a hard-switched converter, a switch turns off while carrying the full
load current $\IL$ (which can be tens or hundreds of amperes).
The current snaps off abruptly and the energy $\frac{1}{2}L_\text{stray}I^2$
stored in stray wiring inductance causes a large voltage spike across the
switch.

Here $S_2$ carries only $i_\text{boost} = \ir - \IL$ at the moment it turns
off.
$i_\text{boost}$ is a \emph{control parameter} that the engineer can keep
small.
Moreover, the current does not snap off into stray inductance --- it
commutates smoothly into $\Cr$, which absorbs it without a spike.
$\vf$ rises at the finite rate $i_\text{boost}/\Cr$ rather than jumping
discontinuously.
This is why the turn-off of $S_2$ is described as
\emph{near-zero-current switching} (near-ZCS): the current being interrupted
is small and the voltage transition is smooth.

% ── State (d) explanation ─────────────────────────────────────────────────────
\subsection*{State (d) — the resonant LC swing: $\vf$ rises from 0 to $\Vdc$}

\figentry{fig:P1Fig7d}{P1_Fig7d_state4_resonant_swing}
    {P1 Fig.\,7(d) — Resonant LC swing.
     $\vf$ rises in an S-curve from 0 to $\Vdc$; $\ir$ peaks then falls.
     $A_2$ remains on; both $\Cr/2$ snubbers are active (labelled).
     No main switch or diode conducts.
     Source: De~Doncker \& Lyons, IEEE IAS 1990, p.\,4--5.}{0.50}

\textbf{State~(d) is the reason the ARCP exists.}

Everything in states (a)--(c) was preparation: ramping $\ir$ to the right
starting value, establishing the right initial voltage.
State~(d) is where those initial conditions are put to work: the resonant
swing carries $\vf$ all the way from 0 to $\Vdc$, enabling $S_1$ to be
turned on at zero voltage.

\medskip
\textbf{Top panel (waveform) — what you see in the state\,(d) strip.}

We are in the fourth strip of the waveform in \figref{fig:P1Fig7d},
labelled \textit{Cr/A2}.
\begin{itemize}
    \item $\vf$ rises in a smooth \emph{S-shaped curve} from 0 towards
          $\Vdc$.  It does not ramp linearly --- it follows a sinusoidal
          shape, slow at first, fastest at the midpoint $\Vdc/2$, then
          slowing again as it approaches $\Vdc$.
    \item $\ir$ first \emph{rises} above its starting value
          $\IL + i_\text{boost}$, reaches a peak, then \emph{falls} back
          down.  The peak occurs when $\vf$ passes through $\Vdc/2$.
\end{itemize}

\medskip
\textbf{Bottom panel (circuit snapshot) — only $A_2$, $\Lr$, and the two
$\Cr/2$ caps.}

Looking at \figref{fig:P1Fig7d}: $A_2$ is the only switch conducting.
Both $\Cr/2$ snubbers are explicitly labelled and shown with bold
connections --- one charging (across $S_1$, from 0 towards $\Vdc$) and one
discharging (across $S_2$, from $\Vdc$ towards 0).
All other devices --- $S_1$, $S_2$, $D_1$, $D_2$ --- are open.

\medskip
\textbf{Why $A_2$ must stay on during the swing.}

This is a subtle but critical point.
If $A_2$ were turned off the moment $S_2$ opened, $\Lr$ would be
disconnected from the DC midpoint.
The LC oscillation would then be a \emph{free} oscillation with its
equilibrium at $\vf = 0$ --- the circuit would simply ring around zero
and $\vf$ would never reach $\Vdc$.

With $A_2$ on, the DC midpoint ($\Vdc/2$) remains connected to $\Lr$
throughout the swing.
This shifts the equilibrium of the oscillation to $\vf = \Vdc/2$ (the
forcing point), so the swing is centred at the midpoint of the bus and
can reach $\Vdc$ on the far side.
This is precisely the \emph{forced harmonic oscillator} we derived in the
green box of Section~\ref{sec:P1Fig5}.

\medskip
\textbf{The governing equations --- already derived in Section~\ref{sec:P1Fig5}.}

Check the three conditions from the green box against the current circuit:
\begin{enumerate}
    \item $A_2$ is \textbf{on} --- DC midpoint connected to $\Lr$. \checkmark
    \item $D_2$ has \textbf{stopped conducting} --- it turned off at the
          end of state~(b). \checkmark
    \item Both $\Cr/2$ caps are \textbf{free} --- $S_2$ opened at the
          start of state~(d). \checkmark
\end{enumerate}
All three conditions are satisfied simultaneously for the first time.
The governing equations \eqref{eq:dvf_dt}--\eqref{eq:dir_dt} therefore
apply exactly here:
\begin{equation*}
    \frac{d\vf}{dt} = \frac{\ir - \IL}{\Cr}, \qquad
    \frac{d\ir}{dt} = \frac{\Vdc/2 - \vf}{\Lr}
\end{equation*}
Combining these (as shown in the green box) gives the forced harmonic
oscillator:
\begin{equation*}
    \ddot{\vf} + \omegar^2\,\vf = \omegar^2\,\frac{\Vdc}{2},
    \qquad \omegar = \frac{1}{\sqrt{\Lr\Cr}},\quad \Zr = \sqrt{\frac{\Lr}{\Cr}}
\end{equation*}

\medskip
\textbf{The solution --- a circular arc in the phase plane.}

As derived in the green box, the trajectory in the
$({\vf},\; j\Zr)$ phase plane is a \textbf{circle} centred at
$(\Vdc/2,\; 0)$ with radius:
\begin{equation*}
    R = \sqrt{\left(\frac{\Vdc}{2}\right)^{\!2} + (i_\text{boost}\,\Zr)^2}
\end{equation*}
The trajectory starts at $(\vf = 0,\; j\Zr = i_\text{boost}\Zr)$ and
travels counter-clockwise until it reaches $\vf = \Vdc$ on the right side
of the circle.
\figref{fig:phase_plane_3ICs} shows this arc for three values of
$i_\text{boost}$.

\medskip
\textbf{Why $\ir$ peaks at $\vf = \Vdc/2$.}

From the KCL equation $d\vf/dt = (\ir - \IL)/\Cr$:
$\vf$ is rising fastest when $\ir - \IL$ is largest, i.e.\ when $\ir$ is
at its peak.
In the phase plane, $\ir$ is maximum when the trajectory passes through
the topmost point of the circle, which occurs at $\vf = \Vdc/2$ (the
centre of the circle lies directly below this point).
After the peak, $\ir$ decreases while $\vf$ continues to rise towards
$\Vdc$.

\medskip
\textbf{What ends this state.}

State~(d) ends the moment $\vf$ reaches $\Vdc$.
At that instant, $D_1$ --- the anti-parallel diode of the upper switch
$S_1$, with its anode at the pole node and cathode at $\Vdc$ --- becomes
forward-biased and starts conducting.
$D_1$ clamps $\vf$ at $\Vdc$, the resonant swing stops, and the circuit
enters state~(e).

(f) & $D_1$, $\Lr$          & $\vf = \Vdc$ (clamped)             & $\ir$ ramps back down: $\dot{\ir} = -\Vdc/2\Lr$ \\
(g) & $S_1$ ON              & $\vf = \Vdc$ (steady)              & $\ir = 0$; aux.\ branch fully off \\
\bottomrule
\end{tabular}
\end{circuitbox}

\begin{eqnbox}
\textbf{State (b) — linear boost ramp:}
\begin{equation}
    \ir(t) = \frac{\Vdc}{2\Lr}\,t, \qquad
    T_\text{boost} = \frac{2\Lr\,i_\text{boost}}{\Vdc}
    \label{eq:T_boost}
\end{equation}

\textbf{State (d) — resonant swing (forced LC oscillator):}
\begin{equation}
    \ddot{\vf} + \omegar^2\,\vf = \omegar^2\cdot\frac{\Vdc}{2},
    \qquad \omegar = \frac{1}{\sqrt{\Lr\Cr}},\quad \Zr = \sqrt{\frac{\Lr}{\Cr}}
    \label{eq:swing_ode}
\end{equation}

Solution with initial conditions $\vf(0)=0$, $\ir(0)=\IL+i_\text{boost}$:
\begin{align}
    \vf(t)  &= \frac{\Vdc}{2}(1 - \cos\omegar t)
               + (\ir(0)-\IL)\,\Zr\sin\omegar t
    \label{eq:vf_swing} \\[4pt]
    \ir(t)  &= \IL + (\ir(0)-\IL)\cos\omegar t
               + \frac{\Vdc/2}{\Zr}\sin\omegar t
    \label{eq:ir_swing}
\end{align}

\textbf{Peak current} (at end of state d, when $\vf = \Vdc$):
\begin{equation}
    I_\text{peak} = \IL + i_\text{boost} + \frac{\Vdc}{2\Zr}
    \label{eq:Ipeak_diode}
\end{equation}

\textbf{Resonant half-swing duration} (state d):
\begin{equation}
    T_\text{res} = \frac{\pi}{\omegar} = \pi\sqrt{\Lr\Cr}
    \label{eq:T_res}
\end{equation}

\textbf{Total commutation time} (states b through f):
\begin{equation}
    T_c = T_\text{boost} + T_\text{res}
        = \frac{2\Lr\,i_\text{boost}}{\Vdc} + \pi\sqrt{\Lr\Cr}
    \label{eq:T_comm_diode}
\end{equation}
\end{eqnbox}

\begin{physbox}
\textbf{Why seven states?} Each state has a distinct set of conducting devices,
so the governing equation changes at each transition.

\textbf{States (a)--(c)} are purely inductive: $\vf$ is pinned to 0 by $D_2$,
so the full $\Vdc/2$ appears across $\Lr$ and drives a linear current ramp.
This is energy storage, not switching.

\textbf{State (d)} is the only state where the LC oscillator is active.
The circular trajectory in the $(\vf,\, \ir-\IL)$ phase plane must reach
$\vf = \Vdc$ for ZVS.  Circle radius:
$R = \sqrt{i_\text{boost}^2 + (\Vdc/2\Zr)^2}$.
ZVS is satisfied when $i_\text{boost} \geq 0$.

\textbf{State (f)} recovers residual energy from $\Lr$ back into the DC bus
through $D_1$ — this is why the auxiliary circuit is lossless in the ideal case.

\textbf{State (g)} is the new idle state. The sequence (a)$\to$(g) is the
complete low-to-high transition; the symmetric high-to-low uses $A_2$ instead
of $A_1$ and is the mirror image.
\end{physbox}
